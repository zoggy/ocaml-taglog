Module Taglog.Ops

module Ops: sig .. end
val ?? : 'a -> 'a Taglog.cond
?? x => Tag x
val (~~) : 'a Taglog.cond -> 'a Taglog.cond
~~ x => Not x
val (||) : 'a Taglog.cond -> 'a Taglog.cond -> 'a Taglog.cond
x || y => Or (x,y)
val (&&) : 'a Taglog.cond -> 'a Taglog.cond -> 'a Taglog.cond
x && y => And (x,y)
